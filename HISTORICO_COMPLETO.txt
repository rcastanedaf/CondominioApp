================================================================================
					  VALIDACIONES IMPLEMENTADAS
================================================================================

POR TIPO DE DATO:

TEXTOS:
  • [Required] para obligatorios
  • [StringLength(min, max)]
  • [RegularExpression] para patrones
  Ejemplos: Nombre, Descripcion, Estado

NUMEROS:
  • [Range(min, max)]
  • Decimales: rango 0.01 a 999999999.99
  • Enteros: rango 1 a int.MaxValue
  Ejemplos: Monto, Cantidad, Id

EMAILS:
  • [EmailAddress]
  Ejemplos: Correo, Email_Contacto

FECHAS:
  • Convertidas a DateOnly (validación automática)
  • Formato: YYYY-MM-DD
  Ejemplos: Fecha_Nacimiento, Fecha_Ingreso, Fecha_Multa

HORAS:
  • Convertidas a TimeSpan
  • Formato: HH:MM:SS
  Ejemplos: Hora_Entrada, Hora_Salida, Hora_Inicio

PATRONES:
  • [RegularExpression] para DPI: solo números
  • [RegularExpression] para Teléfono: solo números
  • [RegularExpression] para Moneda: 3 letras mayúsculas
  Ejemplos: ^\d{1,20}$, ^[A-Z]{3}$

ESTADOS:
  • [Range(0, 1)]
  • Valores: 0 (Inactivo) o 1 (Activo)
  Ejemplos: Activo, Presente, Tiene_Costo

IDENTIFICADORES:
  • [Range(1, int.MaxValue)]
  • No permite 0 ni negativos
  Ejemplos: Id_Residente, Id_Propiedad, Registrado_Por

public class EntityCreateRequest
{
	[Required(ErrorMessage = "Campo requerido")]
	[StringLength(100, MinimumLength = 1)]
	public required string Campo { get; set; }

	[Required(ErrorMessage = "Número requerido")]
	[Range(1, int.MaxValue)]
	public required int Id { get; set; }

	[Required(ErrorMessage = "Monto requerido")]
	[Range(0.01, 999999.99)]
	public required decimal Monto { get; set; }
}

PATRON CONTROLLER:
```csharp
[ApiController]
[Route("[controller]")]
public class EntityController : ControllerBase
{
	private readonly IEntityService _service;
	private readonly ILogger<EntityController> _logger;

	public EntityController(IEntityService service, ILogger<EntityController> logger)
	{
		_service = service;
		_logger = logger;
	}

	[HttpPost("create")]
	public async Task<IActionResult> Create([FromBody] EntityCreateRequest request)
	{
		if (!ModelState.IsValid)
		{
			var errors = ModelState.Values.SelectMany(v => v.Errors)
				.Select(e => e.ErrorMessage).ToList();
			return BadRequest(new ApiResponse<object>
			{
				Success = false,
				Message = "Error en los datos enviados",
				Data = errors
			});
		}

		try
		{
			var result = await _service.CreateAsync(request);
			return Ok(new ApiResponse<object>
			{
				Success = true,
				Message = "Operación exitosa",
				Data = result
			});
		}
		catch (Exception ex)
		{
			_logger.LogError(ex, "Error al crear");
			return BadRequest(new ApiResponse<object>
			{
				Success = false,
				Message = ex.Message,
				Data = null
			});
		}
	}

	[HttpGet("get-id/{id}")]
	public async Task<IActionResult> GetId([FromRoute] int id)
	{
		if (id <= 0)
			return BadRequest(new ApiResponse<object>
			{
				Success = false,
				Message = "El ID debe ser válido",
				Data = null
			});

		try
		{
			var result = await _service.GetByIdAsync(id);
			if (result == null)
				return NotFound(new ApiResponse<object>
				{
					Success = false,
					Message = "No encontrado",
					Data = null
				});

			return Ok(new ApiResponse<object>
			{
				Success = true,
				Message = "Operación exitosa",
				Data = result
			});
		}
		catch (Exception ex)
		{
			_logger.LogError(ex, "Error al obtener");
			return BadRequest(new ApiResponse<object>
			{
				Success = false,
				Message = ex.Message,
				Data = null
			});
		}
	}
}
```
