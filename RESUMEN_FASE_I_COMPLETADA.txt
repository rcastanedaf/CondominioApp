PATRON GET (Por ID):
────────────────────
[HttpGet("get-id-entity/{id}")]
public async Task<IActionResult> GetId([FromRoute] int id)
{
	if (id <= 0)
		return BadRequest(new ApiResponse<object> { ... });

	var result = await _service.GetByIdAsync(id);

	if (result == null)
		return NotFound(new ApiResponse<object> { ... });

	return Ok(new ApiResponse<object> { ... });
}

PATRON POST (Crear):
────────────────────
[HttpPost("create-entity")]
public async Task<IActionResult> Create([FromBody] EntityCreateRequest request)
{
	if (!ModelState.IsValid)
	{
		var errors = ModelState.Values.SelectMany(v => v.Errors)
			.Select(e => e.ErrorMessage).ToList();
		return BadRequest(new ApiResponse<object> { Data = errors });
	}

	try
	{
		var result = await _service.CreateAsync(request);
		return Ok(new ApiResponse<object> { Data = result });
	}
	catch (Exception ex)
	{
		_logger.LogError(ex, "Error al crear");
		return BadRequest(new ApiResponse<object> { Message = ex.Message });
	}
}

PATRON PUT (Actualizar):
─────────────────────────
[HttpPut("update-entity/{id}")]
public async Task<IActionResult> Update([FromRoute] int id, [FromBody] EntityUpdateRequest request)
{
	if (id <= 0)
		return BadRequest(new ApiResponse<object> { ... });

	if (!ModelState.IsValid)
	{
		var errors = ModelState.Values.SelectMany(v => v.Errors)
			.Select(e => e.ErrorMessage).ToList();
		return BadRequest(new ApiResponse<object> { Data = errors });
	}

	try
	{
		var result = await _service.UpdateAsync(id, request);
		return Ok(new ApiResponse<object> { Data = result });
	}
	catch (Exception ex)
	{
		_logger.LogError(ex, "Error al actualizar");
		return BadRequest(new ApiResponse<object> { Message = ex.Message });
	}
}

PATRON DELETE:
───────────────
[HttpDelete("delete-entity/{id}")]
public async Task<IActionResult> Delete([FromRoute] int id)
{
	if (id <= 0)
		return BadRequest(new ApiResponse<object> { ... });

	try
	{
		var result = await _service.DeleteAsync(id);
		return Ok(new ApiResponse<object> { Data = result });
	}
	catch (Exception ex)
	{
		_logger.LogError(ex, "Error al eliminar");
		return BadRequest(new ApiResponse<object> { Message = ex.Message });
	}
}
